#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, q;
    cin >> n >> q;
    string s;
    cin >> s; // or use int array if needed

    // 1. DP table for palindrome
    vector<vector<bool>> dp(n, vector<bool>(n, false));
    for(int i = 0; i < n; i++) dp[i][i] = true;
    for(int i = 0; i < n-1; i++) dp[i][i+1] = (s[i] == s[i+1]);

    for(int len = 3; len <= n; len++) {
        for(int i = 0; i + len -1 < n; i++) {
            int j = i + len -1;
            dp[i][j] = (s[i] == s[j] && dp[i+1][j-1]);
        }
    }

    // 2. Precompute total palindromes in ranges using inclusion-exclusion
    vector<vector<int>> total(n, vector<int>(n, 0));
    for(int i = n-1; i >=0; i--) {
        for(int j = i; j < n; j++) {
            total[i][j] = dp[i][j] ? 1 : 0;
            if(j-1 >= i) total[i][j] += total[i][j-1];
            if(i+1 < n) total[i][j] += total[i+1][j];
            if(i+1 < n && j-1 >= i) total[i][j] -= total[i+1][j-1];
        }
    }

    // 3. Answer queries
    while(q--) {
        int l, r;
        cin >> l >> r;
        l--; r--; // 0-indexed
        cout << total[l][r] << "\n";
    }

    return 0;
}
