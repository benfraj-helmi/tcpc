#include <bits/stdc++.h>
using namespace std;

struct Node {
    Node* child[2];
    Node() {
        child[0] = child[1] = nullptr;
    }
};

struct BinaryTrie {
    Node* root;
    BinaryTrie() { root = new Node(); }

    // Ajouter un nombre
    void insert(int num) {
        Node* node = root;
        for(int i=31;i>=0;i--) {
            int bit = (num >> i) & 1;
            if(!node->child[bit])
                node->child[bit] = new Node();
            node = node->child[bit];
        }
    }

    // Maximum XOR avec num
    int maxXOR(int num) {
        Node* node = root;
        int res = 0;
        for(int i=31;i>=0;i--) {
            int bit = (num >> i) & 1;
            int preferred = 1 - bit; // essayer de choisir le bit opposé pour max XOR
            if(node->child[preferred]) {
                res |= (1 << i);
                node = node->child[preferred];
            } else {
                node = node->child[bit];
            }
        }
        return res;
    }
};

int main() {
    BinaryTrie trie;
    vector<int> arr = {3, 10, 5, 25, 2, 8};
    for(int x : arr) trie.insert(x);

    cout << trie.maxXOR(5) << "\n"; // 28 car 5^25 = 28
    cout << trie.maxXOR(10) << "\n"; // 31 car 10^21 = 31
}
