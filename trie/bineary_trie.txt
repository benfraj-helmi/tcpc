#include <bits/stdc++.h>
using namespace std;

struct Node {
    Node* child[2];
    int cnt; // combien de nombres passent par ce noeud

    Node() {
        child[0] = child[1] = nullptr;
        cnt = 0;
    }
};

struct BinaryTrie {
    Node* root;

    BinaryTrie() { 
        root = new Node(); 
    }

    // Ajouter un nombre
    void insert(int num) {
        Node* node = root;
        for(int i = 31; i >= 0; i--) {
            int bit = (num >> i) & 1;

            if(!node->child[bit])
                node->child[bit] = new Node();

            node = node->child[bit];
            node->cnt++;
        }
    }

    // Supprimer une occurrence
    void removeoneocc(int num) {
        Node* node = root;
        for(int i = 31; i >= 0; i--) {
            int bit = (num >> i) & 1;
            Node* next = node->child[bit];

            if(!next) return; // sécurité

            next->cnt--;

            // Si plus aucun nombre ne passe ici → supprimer
            if(next->cnt == 0) {
                delete next;
                node->child[bit] = nullptr;
                return;
            }

            node = next;
        }
    }

    // Maximum XOR avec num
    int maxXOR(int num) {
        Node* node = root;
        int res = 0;

        for(int i = 31; i >= 0; i--) {
            int bit = (num >> i) & 1;
            int preferred = 1 - bit;

            if(node->child[preferred]) {
                res |= (1 << i);
                node = node->child[preferred];
            } else if(node->child[bit]) {
                node = node->child[bit];
            }
        }

        return res;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    BinaryTrie trie;

    int n;
    cin >> n;

    while(n--) {
        char e;
        int x;
        cin >> e >> x;

        if(e == '+') {
            trie.insert(x);
        }
        else if(e == '-') {
            trie.removeoneocc(x);
        }
        else { // '?'
            cout << trie.maxXOR(x) << "\n";
        }
    }
}
