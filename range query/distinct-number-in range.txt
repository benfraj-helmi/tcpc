#include <bits/stdc++.h>
using namespace std;

const int N = 200005;
int a[N];
int seg[4*N]; // chaque nœud = masque de 26 bits
int n;

// Construire le segment tree
void build(int node, int l, int r) {
    if (l == r) {
        seg[node] = 1 << a[l]; // bit correspondant à a[l]
        return;
    }
    int mid = (l + r) / 2;
    build(node*2, l, mid);
    build(node*2+1, mid+1, r);
    seg[node] = seg[node*2] | seg[node*2+1]; // OR des enfants
}

// Mettre à jour un élément
void update(int node, int l, int r, int pos, int val) {
    if (l == r) {
        seg[node] = 1 << val;
        return;
    }
    int mid = (l + r) / 2;
    if (pos <= mid) update(node*2, l, mid, pos, val);
    else update(node*2+1, mid+1, r, pos, val);
    seg[node] = seg[node*2] | seg[node*2+1];
}

// Requête sur un intervalle [ql, qr]
int query(int node, int l, int r, int ql, int qr) {
    if (qr < l || r < ql) return 0;
    if (ql <= l && r <= qr) return seg[node];
    int mid = (l + r) / 2;
    return query(node*2, l, mid, ql, qr) |
           query(node*2+1, mid+1, r, ql, qr);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;int q; cin >> q;
    for (int i = 1; i <= n; i++) {
        char c; cin >> c;
        a[i] = c - 'a'; 
    }

    build(1, 1, n);

    
    while (q--) {
        int type; cin >> type;
        if (type == 1) {
            int l, r; cin >> l >> r;
            int mask = query(1, 1, n, l, r);
            cout << __builtin_popcount(mask) << '\n'; 
        } else {
            int pos; char c; cin >> pos >> c;
            update(1, 1, n, pos, c - 'a');
        }
    }
}
