long long gcd(long long a, long long b) {
    return b == 0 ? a : gcd(b, a % b);
}

void dfs_lca(int u, int p) {
    up[0][u] = p;
    depth[u] = depth[p] + 1;
    /*
    sumUp[0][u] = val[u];
    minUp[0][u] = val[u];
    maxUp[0][u] = val[u];
    gcdUp[0][u] = val[u];
    xorUp[0][u] = val[u];
    orUp[0][u] = val[u];
    andUp[0][u] = val[u];
    */
    for (auto e : adj[u]) {
        int v = e.to;
        if (v == p) continue;
        dfs_lca(v, u);
    }
}

void build_lca(int root = 1) {
    depth[0] = -1;
    dfs_lca(root, 0);

    for (int k = 1; k < LOG; k++) {
        for (int i = 1; i <= n; i++) {
            int mid = up[k - 1][i];
            up[k][i] = up[k - 1][mid];
             /*
            sumUp[k][i] = sumUp[k - 1][i] + sumUp[k - 1][mid];
            minUp[k][i] = min(minUp[k - 1][i], minUp[k - 1][mid]);
            maxUp[k][i] = max(maxUp[k - 1][i], maxUp[k - 1][mid]);
            gcdUp[k][i] = gcd(gcdUp[k - 1][i], gcdUp[k - 1][mid]);
            xorUp[k][i] = xorUp[k - 1][i] ^ xorUp[k - 1][mid];
            orUp[k][i] = orUp[k - 1][i] | orUp[k - 1][mid];
            andUp[k][i] = andUp[k - 1][i] & andUp[k - 1][mid];
            */
        }
    }
}

int lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);
    int diff = depth[u] - depth[v];
    for (int k = 0; k < LOG; k++)
        if (diff & (1 << k)) u = up[k][u];
    if (u == v) return u;
    for (int k = LOG - 1; k >= 0; k--)
        if (up[k][u] != up[k][v]) { u = up[k][u]; v = up[k][v]; }
    return up[0][u];
}

void dfs_acc(int u, int p) {
    for (auto e : adj[u]) {
        int v = e.to, id = e.id;
        if (v == p) continue;
        dfs_acc(v, u);
        cnt[u] += cnt[v];
        edgeAns[id] = cnt[v];
    }
}

vector< pair< pair<int,int>, long long > >
count_edges_on_paths(const vector<pair<int,int>>& queries) {
    fill(cnt, cnt+n+1, 0);
    fill(edgeAns, edgeAns+n+1, 0);
    for (auto [u, v] : queries) {
        int w = lca(u, v);
        cnt[u]++; cnt[v]++; cnt[w] -= 2;
    }
    dfs_acc(1, 0);

    vector< pair< pair<int,int>, long long > > res;
    for (int i = 1; i <= n-1; i++)
        res.push_back({{eu[i], ev[i]}, edgeAns[i]});
    return res;
}
int dist(int u, int v) {
    int w = lca(u, v);
    return depth[u] + depth[v] - 2 * depth[w];
}

pair<int,int> farthest_pair_in_set(const vector<int>& nodes) {
    if (nodes.size() < 2) return {-1, -1};

    int x = nodes[0];
    int a = x, best = -1;

    for (int u : nodes) {
        int d = dist(x, u);
        if (d > best) {
            best = d;
            a = u;
        }
    }

    int b = a;
    best = -1;
    for (int u : nodes) {
        int d = dist(a, u);
        if (d > best) {
            best = d;
            b = u;
        }
    }

    return {a, b};
}
vector<int> get_path_nodes(int u, int v) {
    int w = lca(u, v);
    vector<int> path1, path2;

    while (u != w) {
        path1.push_back(u);
        u = up[0][u];
    }
    path1.push_back(w);

    while (v != w) {
        path2.push_back(v);
        v = up[0][v];
    }

    reverse(path2.begin(), path2.end());
    for (int x : path2) path1.push_back(x);

    return path1;
}

    }

    for (int i=0;i<q;i++) cin >> queries[i].first >> queries[i].second;

    // edges count
    auto edgeCount = count_edges_on_paths(queries);

    // print edges count
    for(auto &e : edgeCount)
        cout << e.first.first << " " << e.first.second << " : " << e.second << "\n";

    // print query path info
    for(auto [u,v] : queries){

    }

    return 0;
}
